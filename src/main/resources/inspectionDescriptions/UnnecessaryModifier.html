Сообщает о избыточных модификаторах и предлагает их удалить. Итоговый код станет короче, но его поведение и смысл останутся неизменными.

Пример 1:


  // В Java 17 и выше весь код неявно strictfp
  strictfp class X {
    // Внутренние перечисления неявно static
    static enum Inner {
        A, B, C
    }
    // Внутренние записи неявно static
    static record R() {
    }
  }

Пример 2:


  final record R() {
    // Все записи неявно final
  }

Пример 3:


  // Все интерфейсы неявно abstract
  abstract interface Printer {
    // Все члены интерфейса неявно public
    public int size();
    // Все внутренние классы интерфейса неявно static
    static class Inner {}
  }
