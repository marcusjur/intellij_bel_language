Обнаружение неиспользуемых переменных, локальных переменных и источников данных в модулях Terraform с предоставлением быстрого исправления для их удаления.

Пример проблемы:


data "aws_ami" "latest_amazon_linux" {
  most_recent = true
  filter {
    name   = "name"
    values = ["amzn2-ami-hvm-*-x86_64-gp2"]
  }
  owners = ["amazon"]
}
data "aws_vpc" "unused_data_source" {
  default = true
}
resource "aws_instance" "example" {
  ami           = data.aws_ami.latest_amazon_linux.id
  instance_type = "t2.micro"
  tags = {
    Name = "ExampleInstance"
  }
}
variable "used_variable" {
  description = "This variable is used in resource configuration"
  type        = string
  default     = "ami-01456a894f71116f2"
}
locals {
  instance_name = "used-instance"
}
resource "aws_instance" "example1" {
  ami           = var.used_variable
  instance_type = "t2.micro"
  tags = {
    Name = local.instance_name
  }
}
variable "unused_variable1" {
  description = "This variable is not used anywhere in the configuration"
  type        = string
  default     = "default_value1"
}
locals {
  unused_local1 = "This is an unused local value"
}

В этом примере unused_data_source, unused_variable1 и unused_local1 объявлены в файле, но нигде не используются.

После применения быстрого исправления:


data "aws_ami" "latest_amazon_linux" {
  most_recent = true
  filter {
    name   = "name"
    values = ["amzn2-ami-hvm-*-x86_64-gp2"]
  }
  owners = ["amazon"]
}
resource "aws_instance" "example" {
  ami           = data.aws_ami.latest_amazon_linux.id
  instance_type = "t2.micro"
  tags = {
    Name = "ExampleInstance"
  }
}
variable "used_variable" {
  description = "This variable is used in resource configuration"
  type        = string
  default     = "ami-01456a894f71116f2"
}
locals {
  instance_name = "used-instance"
}
resource "aws_instance" "example1" {
  ami           = var.used_variable
  instance_type = "t2.micro"
  tags = {
    Name = local.instance_name
  }
}
locals {
}
