Сообщает о некорректных схемных бинах, определённых в контексте XML-приложения.


  <beans ... xmlns:util="https://www.springframework.org/schema/util">
    <bean class="MyBean" id="myBean"/>
    <bean class="ErrTypeBean" id="errTypeBean"/>
    <util:list list-class="java.lang.String"/> // Сообщает "класс должен быть наследником 'java.util.List'"
    <util:set value-type="MyBean">
      <ref bean="fooBean"/>
      <ref bean="errTypeBean"/> // Сообщает "невозможно добавить в коллекцию типа 'MyBean'"
    </util:set>
  </beans>
