Сообщает о недопустимых типах внедрения (свойства и аргументы конструктора).

Допустимы следующие типы точек внедрения: ref, idref, любые типы коллекций, внутренние Bean-компоненты и т.д.

Пример:


  public class InjectionValueTypes{
  public List<MyBean> setListBeans() {...}
    public MyBean setMyBean() {...}
    }
  }


  <beans>
    <bean class="MyBean" id="myBean"/>
    <bean class="ErrTypeBean" id="errTypeBean"/>
    <bean class="InjectionValueTypes" id="kt">
      <property name="myBean" ref="errTypeBean"/>
      <!-- Сообщение: "Свойство типа 'MyBean' не может быть внедрено из 'ErrTypeBean'" -->
      <property name="listBeans">
        <list>
          <idref local="myBean"/>
          <ref bean="myBean"/>
          <ref bean="errTypeBean"/> <!-- Сообщение: "Невозможно добавить в коллекцию типа 'MyBean'" -->
          <bean class="ErrTypeBean"/> <!-- Сообщение: "Bean должен быть типа 'MyBean'" -->
        </list>
      </property>
    </bean>
  </beans>
