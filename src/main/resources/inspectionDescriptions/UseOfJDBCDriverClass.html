Сообщает об использовании конкретных классов драйверов JDBC. Использование таких классов привязывает ваш проект к определённой базе данных и драйверу, что противоречит назначению JDBC и приводит к потере переносимости.

Пример:


  import java.sql.Driver;
  abstract class Sample implements Driver {
    public void foo() {
        Sample sample;
    }
  }
