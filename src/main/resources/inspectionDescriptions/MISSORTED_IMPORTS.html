Сообщает об операторах import, которые не соответствуют текущему стилю кода (см. "Настройки|Редактор|Стиль кода").

Пример:


  import java.util.List;
  import java.util.ArrayList;
  public class Example {
    List list = new ArrayList();
  }

После применения быстрого исправления "Оптимизовать импорты": 


  import java.util.ArrayList;
  import java.util.List;
  public class Example {
    List list = new ArrayList();
  }
