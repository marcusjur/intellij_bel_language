Конструктор отчётов не может сгенерировать соответствующий запрос для метода репозитория Jakarta Data.

Пример:


interface PersonRepository extends Repository<Person, Long> {
  List<Person> findByEmailAndLastname(String email, String lastname);
  List<Person> findByUnknownAndLastname(EmailAddress emailAddress, String lastname); // Не удаётся разрешить свойство 'Unknown'
  List<Person> findByAndLastname(String email, String lastname); // Отсутствует имя свойства
  List<Person> findByLastnameOrderByFirstnameAsc(String lastname);
  List<Person> findByLastnameOrderByUnknownDesc(String lastname);  // Не удаётся разрешить свойство 'Unknown'
}
