Сообщает о точках внедрения в классах Bean с отсутствующими или неоднозначными зависимостями.

Пример:


  interface Engine {}
  @Singleton
  class GasEngine implements Engine {}
  @Singleton
  class ElectricEngine implements Engine {}
  @Singleton
  class Car {
    @Inject
    Engine engine; // Неоднозначная зависимость: несколько Bean соответствуют точке внедрения
  }
