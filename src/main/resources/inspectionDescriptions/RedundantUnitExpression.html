Обнаружение избыточных выражений Unit.

В Kotlin Unit может использоваться как возвращаемый тип для функций, которые не возвращают значимого результата. Тип Unit может иметь только одно возможное значение — объект Unit.

Пример:


  fun redundantA(): Unit {
      return Unit // Избыточно, 'Unit' возвращается по умолчанию и соответствует ожидаемому возвращаемому типу
  }
  fun requiredA(condition: Boolean): Any {
      if (condition) return "hello"
      return Unit // Явное указание 'Unit' необходимо, так как ожидаемый тип — 'Any'
  }
  fun redundantB(condition: Boolean): Any = if (condition) {
      fun ancillary(): Int = 1
      println("${ancillary()}")
      Unit // Избыточно, так как предыдущее выражение уже имеет тип 'Unit'
  } else {
      println("else")
  }
  fun requiredB(condition: Boolean): Any = if (condition) {
      1024
      Unit // Необходимо, иначе '1024' (Int) будет возвращаемым значением
  } else {
      println("else")
  }
