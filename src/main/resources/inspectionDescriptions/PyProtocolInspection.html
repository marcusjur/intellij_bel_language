Сообщает о недопустимых определениях и использовании протоколов, введенных в PEP-544.

Пример:


from typing import Protocol
class MyProtocol(Protocol):
    def method(self, p: int) -> str:
        pass
class MyClass(MyProtocol):
    def method(self, p: str) -> int: # тип 'method' несовместим с 'MyProtocol'
        pass
class MyAnotherProtocol(MyClass, Protocol): # все базовые классы протокола должны быть протоколами
    pass
