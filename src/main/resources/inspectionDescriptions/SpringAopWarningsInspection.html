Сообщение о некорректных аспектах и точках внедрения. При сопоставлении некоторых AOP-рекомендаций с бинами аспектов могут возникнуть ошибки времени выполнения.

Пример: 


<beans>
  <aop:config>
    <aop:pointcut expression="args()"/> <!-- Точка внедрения должна иметь ID -->
    <aop:aspect/> <!-- Должен быть определен атрибут 'ref' -->
  </aop:config>
</beans>
<beans>
  <bean id="aspect1" class="Aspect1"/>
  <bean id="aspect2" class="Aspect2"/>
  <aop:config>
    <aop:aspect ref="aspect1"> <!-- Некоторые рекомендации сопоставляются с этим бином -->
    <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
    <aop:aspect ref="aspect2">
      <aop:before method="xxx" pointcut="execution(* foo())"/>
    </aop:aspect>
  </aop:config>
</beans>
