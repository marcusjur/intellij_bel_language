Обнаружен конфликт в конфигурации атрибутов <bean>.

Пример:


  <beans>
    <bean class="java.lang.String" factory-bean="factoryBean"/>
    <!-- Требуется только один из атрибутов: 'class' или 'factory-bean' -->
    <bean factory-bean="factoryBean" id="missingFactoryMethod"/> <!-- При наличии factory-bean требуется factory-method -->
    <!-- Атрибут abstract со значением по умолчанию 'false' является избыточным -->
    <bean abstract="false" id="superfluousDefaultAttributeAbstract"/>
    <!-- Атрибут scope со значением по умолчанию 'singleton' является избыточным -->
    <bean id="superfluousDefaultAttributeScope" scope="singleton"/>
    <bean class="java.lang.StringBuilder">
      <property name="length">
        <!-- Атрибуты ID, name и scope являются избыточными -->
        <bean id="beanId" name="beanName" scope="prototype"/>
      </property>
    </bean>
  </beans>
