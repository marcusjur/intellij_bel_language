Сообщает о некорректных ссылках на Bean в параметре scheduler аннотации @Scheduled

Пример:


  @Component public class UserController {...}
  @Component public class UserServiceAsyncTaskScheduler extends SimpleAsyncTaskScheduler {...}
  @Component public class UserService {
    @Scheduled(scheduler="userServiceAsyncTaskScheduler", cron="1 * * * * * ?")
    public void scheduleTask() {...}
    @Scheduled(scheduler="userController", cron="1 * * * * * ?") // Сообщение "Bean должен быть одного из следующих типов: TaskScheduler, ScheduledExecutorService"
    public void scheduleTask_2() {...}
    @Scheduled(scheduler="unknown", cron="1 * * * * * ?")   // Сообщение "Невозможно разрешить Bean 'unknown'"
    public void scheduleTask_3() {...}
  }
