Сообщает о неинициализованных или не autowired свойствах сеттеров Bean с аннотацией @Required.

Пример:


<beans>
  <bean class="beans.MyComponent" id="foo"/>
  <!-- Сообщение "Отсутствует обязательное свойство: 'port'" -->
</beans>


public class MyComponent {
   @Required
   public void setPort(int port) // Сообщение "Не внедрено обязательное свойство 'port'"
   {...}
}
