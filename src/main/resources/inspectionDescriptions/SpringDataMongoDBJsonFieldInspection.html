Сообщает о нераспознанных полях документа, указанных в аннотации org.springframework.data.mongodb.repository.Query.

Пример:


@Document
class User {
  @Field
  String name;
}
interface UserRepository extends Repository<User, Long> {
  @Query("{ age: { $gt: ?0 } }") // 'age' подсвечивается как нераспознанное
  List<User> findUsersOlderThan(Integer age);
}
