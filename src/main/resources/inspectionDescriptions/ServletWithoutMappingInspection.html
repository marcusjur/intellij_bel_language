Сообщает о сервлетах без соответствующих привязок.

Пример:


  <web-app>
    <servlet>
      <servlet-name>foo</servlet-name> // Ошибка: сервлет должен иметь привязку
      <servlet-class>javax.servlet.Servlet</servlet-class>
    </servlet>
    <filter>
      <filter-name>bar</filter-name> // Ошибка: фильтр должен иметь привязку
      <filter-class>java.lang.String</filter-class>
    </filter>
  </web-app>
