Сообщает об использовании this в замыканиях для ссылки на свойства внешнего контекста.

Пример:


function Outer() {
  this.outerProp = 1;
  function inner() {
    // Плохая практика, так как 'outerProp' класса Outer
    // не будет обновлен должным образом
    // при вызове 'new Outer()'
    this.outerProp = 2;
  }
  inner();
}
