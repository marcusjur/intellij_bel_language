잘못된 저장소 메서드 매개변수 타입을 보고합니다.

예:


  public class Person {
    private int id;
    @Column
    private String lastname;

    @Column
    private int age;
  }

  public interface CustomerRepository extends Repository<Person, Integer> {
    List<Person> findByAddress(int lastname);  // 'String' 타입 필요
    List<Person> deleteByAge(String str);   // 'int' 타입 필요
  }
