Обнаружение избыточных операторов import.

Неиспользуемые и стандартные импорты можно безопасно удалить.

Пример:


  import kotlin.*
  import kotlin.collections.*
  import kotlin.comparisons.*
  import kotlin.io.*
  import kotlin.ranges.*
  import kotlin.sequences.*
  import kotlin.text.*
  // Специфично для jvm
  import java.lang.*
  import kotlin.jvm.*
  // Специфично для js
  import kotlin.js.*
  import java.io.* // Этот импорт не используется и может быть удален
  import java.util.*
  fun foo(list: ArrayList<String>) {
      list.add("")
  }
