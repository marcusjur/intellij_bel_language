Сообщает о неправильных типах параметров в методах Spring Data CRUD репозитория.

Пример:


  public class Person {
    private int id;
    private String lastname;
    private Address address;
  }
  public class Address {
    private int zipCode;
  }
  public interface CustomerRepository extends CrudRepository<Person, Integer> {
    List<Person> findByAddress(String lastname);  // должен быть тип 'Address'
    List<Person> findByAddressZipCode(String str);   // должен быть тип 'int'
  }
