Сообщение об использовании рекурсивного метода VirtualFile.getChildren().

Это может привести к бесконечному циклу при итерации по циклическим символическим ссылкам. Используйте вместо этого VfsUtilCore.visitChildrenRecursively(). 


void processDirectory(VirtualFile dir) {
  for (VirtualFile file : dir.getChildren()) { // Неверно
    if (!file.isDirectory()) {
      processFile(file);
    } else {
      processDirectory(file); // Рекурсивный вызов
    }
  }
}



void processDirectory(VirtualFile dir) {
  VfsUtilCore.visitChildrenRecursively(dir, new VirtualFileVisitor<Void>() { // Верно
    @Override
    public boolean visitFile(@NotNull VirtualFile file) {
      if (!file.isDirectory()) {
        processFile(file);
      }
      return true;
    }
  });
}
