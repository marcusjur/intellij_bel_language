Сообщение о некорректном использовании <lookup-method> для Bean-компонентов в контексте XML-приложения.

Пример:


  public abstract class FooLookupBean {
    protected abstract FooBean createCommand();
    private FooBean createCommand() {...}
    protected static FooBean createCommandStatic() {...}
    protected abstract FooBean createWithArgs(String foo);
    protected abstract OtherBean createOtherBean();
  }
  

  <beans>
    <bean class="FooLookupBean" id="lookupTest">
      <lookup-method bean="fooBean" name="createCommand"/>
      <!-- private -->
      <lookup-method bean="fooBean" name="createCommand"/> <!-- "Метод должен быть public или protected" -->
      <!-- static -->
      <lookup-method bean="fooBean" name="createCommandStatic"/> <!-- Статические методы здесь не допускаются -->
      <!-- with arguments -->
      <lookup-method bean="fooBean" name="createWithArgs"/> <!-- Метод не должен иметь параметров -->
      <!-- not exist -->
      <lookup-method bean="fooBean" name="unknown"/> <!-- Неизвестный метод -->
      <!-- mismatch -->
      <lookup-method bean="fooBean" name="createOtherBean"/> <!-- Тип возвращаемого значения метода lookup не соответствует типу Bean-компонента 'FooBean' -->
    </bean>
    <bean class="FooBean" id="fooBean"/>
    <bean class="OtherBean" id="otherBean"/>
  </beans>
