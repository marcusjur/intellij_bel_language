Сообщает о переменных шаблона, которые имеют то же имя, что и поля окружающего класса. Из-за такого именования вы можете случайно использовать переменную шаблона вместо поля, даже если плановали использовать одноименное поле.

Рекомендуется переименовать переменную с помощью быстрого исправления.

Пример:


  class Pointless {
    Point p = new Point();
    public void test(Object a) {
      if (a instanceof Point p) {
        System.out.print("a is a point (" + p.x + ", " + p.y + ')');
      } else {
        System.out.print("p is a point (" + p.x + ", " + p.y + ')');
      }
    }
  }
