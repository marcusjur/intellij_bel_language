Обнаружены следующие нарушения спецификации EJB:
  • Метод create() определен некорректно
  • Метод finder() не разрешен

Пример:


public interface SessionHome extends javax.ejb.EJBHome {
  Session create() throws java.rmi.RemoteException; // Ошибка: также должен выбрасывать `javax.ejb.CreateException`
}
