Сообщение о проблемах в контексте приложения на базе XML для Spring Integration:

  • Неразрешённые ссылки на Bean
  • Отсутствующие обязательные теги или атрибуты
  • Неверный тип свойства
  • Несоответствие свойств типа enum
  • Неверный тип ссылочного Bean

Пример:


  <beans ... >
  <int:channel id="channel"> <!-- 'dispatcher' и 'queue' не могут использоваться вместе  -->
    <int:dispatcher failover="true"/>
    <int:priority-queue capacity="4"/>
  </int:channel>
  <int:channel id="another" >
    <int:queue <!--Требуется ровно один из атрибутов: 'message-store' или 'ref'  -->
              capacity="4" message-store="myMessageStore" ref="myQueue"/>
  </int:channel>
   <int:gateway id="asyncGatewayError"
                async-executor="dummyBean"/> <!--Bean должен быть типа 'java.util.concurrent.Executor'  -->
   <int:delayer default-delay="3"
              message-store="myMessageStore"
              scheduler="scheduler"
              wait-for-tasks-to-complete-on-shutdown="false"/>
  <!--Bean должен быть типа 'java.util.concurrent.Executor'  -->
  <int:delayer
    default-delay="INVALID_VALUE"  <!--Невозможно преобразовать строку 'INVALID_VALUE' в целевой класс 'java.lang.Integer' -->
               message-store="dummyBean" <!-- Bean должен быть типа 'org.springframework.integration.store.MessageStore' -->
               scheduler="dummyBean" <!-- Bean должен быть одного из следующих типов: java.util.concurrent.ScheduledExecutorService,org.springframework.scheduling.TaskScheduler-->
               wait-for-tasks-to-complete-on-shutdown="INVALID_VALUE"/> <!-- Невозможно разрешить символ 'INVALID_VALUE' -->
  <bean id="dummyBean" class="java.lang.String"/>
  <bean id="myMessageStore" class="org.springframework.integration.store.SimpleMessageStore"/>
  <task:scheduler id="scheduler"/>
</beans>
