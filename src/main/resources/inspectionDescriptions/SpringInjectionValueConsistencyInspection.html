Сообщает о несогласованных тегах и атрибутах в точках внедрения в контексте XML-приложения.

  • <property> и <constructor-arg> должны указывать ref, value или дочерний элемент
  • <property> и <constructor-arg> не могут содержать несколько дочерних элементов
  • <ref> может содержать атрибут bean, атрибут local или атрибут parent
  • <ref> должен указывать на Bean
  • <idref> может содержать атрибут bean или атрибут local
  • <idref> должен указывать на Bean
  • <map> должен указывать ключ
  • <idref> может содержать атрибут key, атрибут key-ref или дочерний элемент <key>

Пример:


  <beans>
  <!-- 1. Свойство может содержать только атрибут "ref" или атрибут "value" или дочерние элементы -->
  <bean class="MyClass">
    <property name="bean" ref="myBean" value="foo"/>
    <!-- Свойство может содержать только атрибут "ref" или атрибут "value" или дочерние элементы -->
  </bean>
  <!-- 2. Свойство не должно содержать несколько дочерних элементов -->
  <bean class="MyFactory">
    <property name="bean">
    <ref>bean</ref> <!-- Свойство не должно содержать несколько дочерних элементов -->
     <value>foo</value> <!-- Свойство не должно содержать несколько дочерних элементов -->
     </property>
  </bean>
  <!-- 3. Элемент отображения должен указывать ключ -->
  <bean class="MyBean">
     <property name="map">
        <map>
           <entry value="foo"/>  <!-- Элемент отображения должен указывать ключ -->
        </map>
     </property>
  </bean>
  </beans>
