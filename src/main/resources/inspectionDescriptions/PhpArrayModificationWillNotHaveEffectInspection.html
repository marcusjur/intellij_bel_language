Сообщает о вызове функции, которая возвращает массив по значению (что делает операцию присваивания неэффективной). Следует изменить на возврат по ссылке.

Пример:


<?php
class A {
    private $values = [];
    public function getValues() {
        return $this->values;
    }
}
$config = new A();
$config->getValues()['a'] = 'a';
