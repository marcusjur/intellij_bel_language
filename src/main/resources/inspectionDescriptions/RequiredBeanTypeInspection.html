Сообщает о неправильном типе внедрения пользовательских бинов схемы, определённых в контексте XML-приложения.

Пример:


  public class SimpleBean{}


  <beans ...>
    <bean class="SimpleBean" id="simpleBean"/>
       <mvc:annotation-driven
         conversion-service="simpleBean"      <!-- reports "Bean must be of 'org.springframework.core.convert.ConversionService' type" -->
         message-codes-resolver="simpleBean"  <!-- Сообщает "Бин должен быть типа 'org.springframework.validation.MessageCodesResolver'" -->
         validator="simpleBean"               <!-- Сообщает "Бин должен быть типа 'org.springframework.validation.Validator'" -->
    />
  </beans>
