Сообщает о повторяющихся именах Bean-компонентов, определённых в контексте приложения.

Пример:


  <beans profile="PROD">
    <bean class="..." id="duplicatedName"/>  <!-- ошибка -->
    <bean class="..." name="foo_name, duplicatedName"/> <!-- 'duplicatedName' ошибка -->
    <bean class="..." id="toBeAliased"/>
    <alias alias="duplicatedName" name="toBeAliased"/> <!-- 'duplicatedName' ошибка -->
  </beans>
