Отчёт об устаревших XML-элементах в XML-контексте приложения текущей версии Spring Integration.

Для получения дополнительной информации см. руководство по миграции.

Пример:


<beans ... >
   <int:channel id="failChannel"
                dispatcher="failover"/> <!-- Этот атрибут устарел, используйте вместо него подэлемент dispatcher -->
   <int:poller default="true">
     <int:cron-trigger  expression="5"/>  <!-- Этот элемент удален в Spring Integration 2.1, используйте атрибут 'cron' -->
     <int:interval-trigger interval="3"/> <!-- Этот элемент удален в Spring Integration 2.1, используйте атрибут interval-trigger -->
  </int:poller>
  <ftp:inbound-channel-adapter session-factory="ftpSessionFactory"
                               cache-sessions="false" <!--  This attribute is deprecated, use CachingSessionFactory instead -->
                               channel="channel"/>
</beans>
