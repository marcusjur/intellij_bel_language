Сообщать о любых цепочках операторов if-else, где все условия являются выражениями instanceof или сравнениями с классами (например, с String.class). Такие структуры обычно указывают на неудачный объектно-ориентированный дизайн, где подобная диспетчеризация на основе типов должна выполняться через полиморфные вызовы методов, а не через явные цепочки проверок типов.

Пример:


  double getArea(Shape shape) {
    // Предупреждение: неудачная абстракция.
    // Лучше объявить абстрактный метод getArea()
    // в интерфейсе shape
    // и реализовать его в каждом наследнике.
    if (shape instanceof Point) {
      return 0;
    }
    if (shape instanceof Circle) {
      return Math.PI *
        Math.pow(((Circle) shape).radius(), 2);
    }
    if (shape instanceof Rectangle) {
      return ((Rectangle) shape).width() *
        ((Rectangle) shape).height();
    }
    throw new IllegalArgumentException();
  }
