Сообщение о проблемах в XML-контексте Spring Webflow:

  • Неразрешённые ссылки на Bean
  • Отсутствующие обязательные теги или атрибуты
  • Неверные типы атрибутов
  • Несоответствие атрибутов типа enum
  • Неверные типы ссылок на Bean

Пример:


  <beans ...
    xmlns:webflow="http://www.springframework.org/schema/webflow-config ...">
    <webflow:flow-executor id="flowExecutorWithoutFlowRegistryRef"/> <!-- Не найден <flow-registry> с ID по умолчанию 'flowRegistry' -->
    <webflow:flow-executor
      id=""  <!--Значение не должно быть пустым-->
      flow-registry="INVALID_VALUE"  <!-- Невозможно разрешить Bean INVALID_VALUE -->
           <webflow:flow-execution-repository
                   conversation-manager="dummyBean" <!-- Bean должен быть типа 'org.springframework.webflow.conversation.ConversationManager' -->
                    max-executions="INVALID_VALUE" <!-- Значение должно быть целым числом -->
                    max-execution-snapshots="INVALID_VALUE"/>  <!-- Значение должно быть целым числом -->
           <webflow:flow-execution-listeners>
			<webflow:listener criteria="*"
                           ref="dummyBean" /> <!-- Bean должен быть типа 'org.springframework.webflow.execution.FlowExecutionListener' -->
           </webflow:flow-execution-listeners>
    </webflow:flow-executor>
    <webflow:flow-executor id="dummy"
      flow-registry="dummyBean"/> <!-- Bean должен быть типа 'org.springframework.webflow.definition.registry.FlowDefinitionRegistry' -->
    <bean id="dummyBean" class="java.lang.String"/>
  </beans>
