В случае возникновения org.testng.TestNGException, сообщается о наличии поставщиков данных TestNG с одинаковыми именами.

Пример:


  public class DuplicatedDataProviders {
    @DataProvider
    public Object[][] intTestData() { // Дубликат 1
      return new Integer[][]{
        new Integer[]{1, 1},
      };
    }
    @DataProvider(name = "intTestData")
    public Object[][] someTestData() { // Дубликат 2
      return new Integer[][]{
        new Integer[]{1, 1},
      };
    }
    @Test(dataProvider = "intTestData")
    public void testIsOK(Integer key, Integer value) {
      assertEquals(key, value);
    }
  }
