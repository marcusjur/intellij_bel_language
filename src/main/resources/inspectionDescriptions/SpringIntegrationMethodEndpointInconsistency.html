Обнаружено некорректное объявление метода конечной точки Spring Integration.

Пример:


class MyEndpoints {
  @InboundChannelAdapter("channel")
  public void cannotBeVoid() {...} // Метод с аннотацией @InboundChannelAdapter должен иметь возвращаемый тип
  @InboundChannelAdapter("channel")
  public String cannotHaveParams(String s) {..} // Метод с аннотацией @InboundChannelAdapter не может иметь параметров
  @Filter(inputChannel = "channel", // Конечная точка может иметь только один поллер
    outputChannel = "channel2",
    poller = {@Poller(fixedDelay = "100"), @Poller(fixedRate = "100")})
  public void testMultiplePollers() {
  }
  @Filter(inputChannel = "channel",
  outputChannel = "channel2",
  poller = @Poller(value = "poller", maxMessagesPerPoll = "100"))
  public void testValue() {
  }
  @Filter(inputChannel = "channel",
    outputChannel = "channel2",
    poller = @Poller(trigger = "trigger", cron = "0 */10 * * * *")) // Атрибут 'trigger' несовместим с другими атрибутами
  public void testTrigger() {
  }
}
