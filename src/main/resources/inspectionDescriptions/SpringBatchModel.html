Сообщение о проблемах в контексте приложения на основе XML для Spring Batch:
  • Неразрешённые ссылки на Bean
  • Отсутствующие теги и атрибуты
  • Неверные типы свойств
  • Несоответствие свойств типа enum
  • Неверные типы для ссылок на Bean

Пример:


  <beans ... >
  <batch:job id="" <!-- Value must not be empty -->
    incrementer="dummyBean"  <!-- Bean должен быть типа 'org.springframework.batch.core.JobParametersIncrementer' -->
    job-repository="transactionManager"  <!-- Bean должен быть типа 'org.springframework.batch.core.repository.JobRepository' -->
    parent="INVALID_VALUE" <!-- Невозможно разрешить задачу INVALID_VALUE -->
    restartable="INVALID_VALUE"> <!-- Невозможно разрешить символ INVALID_VALUE -->
    <batch:validator
      ref="dataSource">  <!-- Bean должен быть типа 'org.springframework.batch.core.JobParametersValidator' -->
    </batch:validator>
  </batch:job>
  <batch:job-repository/>
  <bean id="dataSource" class="org.springframework.jdbc.datasource.DelegatingDataSource"/>
  <bean id="transactionManager" class="org.springframework.jdbc.datasource.DataSourceTransactionManager"/>
  <bean id="dummyBean" class="java.lang.String"/>
</beans>
